481 template<typename _Tp, typename Functor> inline
482 void Mat::forEach_impl(const Functor& operation) {
483  if (false) {
484  operation(*reinterpret_cast<_Tp*>(0), reinterpret_cast<int*>(0));
485  // If your compiler fail in this line.
486  // Please check that your functor signature is
487  // (_Tp&, const int*) <- multidimential
488  // or (_Tp&, void*) <- in case of you don't need current idx.
489  }
490 
491  CV_Assert(this->total() / this->size[this->dims - 1] <= INT_MAX);
492  const int LINES = static_cast<int>(this->total() / this->size[this->dims - 1]);
493 
494  class PixelOperationWrapper :public ParallelLoopBody
495  {
496  public:
497  PixelOperationWrapper(Mat_<_Tp>* const frame, const Functor& _operation)
498  : mat(frame), op(_operation) {}
499  virtual ~PixelOperationWrapper(){}
500  // ! Overloaded virtual operator
501  // convert range call to row call.
502  virtual void operator()(const Range &range) const {
503  const int DIMS = mat->dims;
504  const int COLS = mat->size[DIMS - 1];
505  if (DIMS <= 2) {
506  for (int row = range.start; row < range.end; ++row) {
507  this->rowCall2(row, COLS);
508  }
509  } else {
510  std::vector<int> idx(COLS);
511  idx[DIMS - 2] = range.start - 1;
512 
513  for (int line_num = range.start; line_num < range.end; ++line_num) {
514  idx[DIMS - 2]++;
515  for (int i = DIMS - 2; i >= 0; --i) {
516  if (idx[i] >= mat->size[i]) {
517  idx[i - 1] += idx[i] / mat->size[i];
518  idx[i] %= mat->size[i];
519  continue; // carry-over;
520  }
521  else {
522  break;
523  }
524  }
525  this->rowCall(&idx[0], COLS, DIMS);
526  }
527  }
528  }
529  private:
530  Mat_<_Tp>* const mat;
531  const Functor op;
532  // ! Call operator for each elements in this row.
533  inline void rowCall(int* const idx, const int COLS, const int DIMS) const {
534  int &col = idx[DIMS - 1];
535  col = 0;
536  _Tp* pixel = &(mat->template at<_Tp>(idx));
537 
538  while (col < COLS) {
539  op(*pixel, const_cast<const int*>(idx));
540  pixel++; col++;
541  }
542  col = 0;
543  }
544  // ! Call operator for each elements in this row. 2d mat special version.
545  inline void rowCall2(const int row, const int COLS) const {
546  union Index{
547  int body[2];
548  operator const int*() const {
549  return reinterpret_cast<const int*>(this);
550  }
551  int& operator[](const int i) {
552  return body[i];
553  }
554  } idx = {{row, 0}};
555  // Special union is needed to avoid
556  // "error: array subscript is above array bounds [-Werror=array-bounds]"
557  // when call the functor `op` such that access idx[3].
558 
559  _Tp* pixel = &(mat->template at<_Tp>(idx));
560  const _Tp* const pixel_end = pixel + COLS;
561  while(pixel < pixel_end) {
562  op(*pixel++, static_cast<const int*>(idx));
563  idx[1]++;
564  }
565  }
566  PixelOperationWrapper& operator=(const PixelOperationWrapper &) {
567  CV_Assert(false);
568  // We can not remove this implementation because Visual Studio warning C4822.
569  return *this;
570  }
571  };
572 
573  parallel_for_(cv::Range(0, LINES), PixelOperationWrapper(reinterpret_cast<Mat_<_Tp>*>(this), operation));
574 }

956 template<typename _Tp, size_t fixed_size> inline void
957 AutoBuffer<_Tp, fixed_size>::resize(size_t _size)
958 {
959  if(_size <= sz)
960  {
961  sz = _size;
962  return;
963  }
964  size_t i, prevsize = sz, minsize = MIN(prevsize, _size);
965  _Tp* prevptr = ptr;
966 
967  ptr = _size > fixed_size ? new _Tp[_size] : buf;
968  sz = _size;
969 
970  if( ptr != prevptr )
971  for( i = 0; i < minsize; i++ )
972  ptr[i] = prevptr[i];
973  for( i = prevsize; i < _size; i++ )
974  ptr[i] = _Tp();
975 
976  if( prevptr != buf )
977  delete[] prevptr;
978 }

1006 // Basic Node class for tree building
1007 template<class OBJECT>
1008 class CV_EXPORTS Node
1009 {
1010 public:
1011  Node()
1012  {
1013  m_pParent = 0;
1014  }
1015  Node(OBJECT& payload) : m_payload(payload)
1016  {
1017  m_pParent = 0;
1018  }
1019  ~Node()
1020  {
1021  removeChilds();
1022  if (m_pParent)
1023  {
1024  int idx = m_pParent->findChild(this);
1025  if (idx >= 0)
1026  m_pParent->m_childs.erase(m_pParent->m_childs.begin() + idx);
1027  }
1028  }
1029 
1030  Node<OBJECT>* findChild(OBJECT& payload) const
1031  {
1032  for(size_t i = 0; i < this->m_childs.size(); i++)
1033  {
1034  if(this->m_childs[i]->m_payload == payload)
1035  return this->m_childs[i];
1036  }
1037  return NULL;
1038  }
1039 
1040  int findChild(Node<OBJECT> *pNode) const
1041  {
1042  for (size_t i = 0; i < this->m_childs.size(); i++)
1043  {
1044  if(this->m_childs[i] == pNode)
1045  return (int)i;
1046  }
1047  return -1;
1048  }
1049 
1050  void addChild(Node<OBJECT> *pNode)
1051  {
1052  if(!pNode)
1053  return;
1054 
1055  CV_Assert(pNode->m_pParent == 0);
1056  pNode->m_pParent = this;
1057  this->m_childs.push_back(pNode);
1058  }
1059 
1060  void removeChilds()
1061  {
1062  for(size_t i = 0; i < m_childs.size(); i++)
1063  {
1064  m_childs[i]->m_pParent = 0; // avoid excessive parent vector trimming
1065  delete m_childs[i];
1066  }
1067  m_childs.clear();
1068  }
1069 
1070  int getDepth()
1071  {
1072  int count = 0;
1073  Node *pParent = m_pParent;
1074  while(pParent) count++, pParent = pParent->m_pParent;
1075  return count;
1076  }
1077 
1078 public:
1079  OBJECT m_payload;
1080  Node<OBJECT>* m_pParent;
1081  std::vector<Node<OBJECT>*> m_childs;
1082 };
