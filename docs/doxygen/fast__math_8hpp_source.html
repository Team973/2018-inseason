92 CV_INLINE int
93 cvRound( double value )
94 {
95 #if ((defined _MSC_VER && defined _M_X64) || (defined __GNUC__ && defined __x86_64__ \
96  && defined __SSE2__ && !defined __APPLE__)) && !defined(__CUDACC__)
97  __m128d t = _mm_set_sd( value );
98  return _mm_cvtsd_si32(t);
99 #elif defined _MSC_VER && defined _M_IX86
100  int t;
101  __asm
102  {
103  fld value;
104  fistp t;
105  }
106  return t;
107 #elif ((defined _MSC_VER && defined _M_ARM) || defined CV_ICC || \
108  defined __GNUC__) && defined HAVE_TEGRA_OPTIMIZATION
109  TEGRA_ROUND_DBL(value);
110 #elif defined CV_ICC || defined __GNUC__
111 # if CV_VFP
112  ARM_ROUND_DBL(value);
113 # else
114  return (int)lrint(value);
115 # endif
116 #else
117  /* it's ok if round does not comply with IEEE754 standard;
118  the tests should allow +/-1 difference when the tested functions use round */
119  return (int)(value + (value >= 0 ? 0.5 : -0.5));
120 #endif
121 }

131 CV_INLINE int cvFloor( double value )
132 {
133 #if (defined _MSC_VER && defined _M_X64 || (defined __GNUC__ && defined __SSE2__ && !defined __APPLE__)) && !defined(__CUDACC__)
134  __m128d t = _mm_set_sd( value );
135  int i = _mm_cvtsd_si32(t);
136  return i - _mm_movemask_pd(_mm_cmplt_sd(t, _mm_cvtsi32_sd(t,i)));
137 #elif defined __GNUC__
138  int i = (int)value;
139  return i - (i > value);
140 #else
141  int i = cvRound(value);

143  return i - (diff < 0);
144 #endif
