170 template<typename T>
171 class WImage
172 {
173 public:
174  typedef T BaseType;
175 
176  // WImage is an abstract class with no other virtual methods so make the
177  // destructor virtual.
178  virtual ~WImage() = 0;
179 
180  // Accessors
181  IplImage* Ipl() {return image_; }
182  const IplImage* Ipl() const {return image_; }
183  T* ImageData() { return reinterpret_cast<T*>(image_->imageData); }
184  const T* ImageData() const {
185  return reinterpret_cast<const T*>(image_->imageData);
186  }
187 
188  int Width() const {return image_->width; }
189  int Height() const {return image_->height; }
190 
191  // WidthStep is the number of bytes to go to the pixel with the next y coord
192  int WidthStep() const {return image_->widthStep; }
193 
194  int Channels() const {return image_->nChannels; }
195  int ChannelSize() const {return sizeof(T); } // number of bytes per channel
196 
197  // Number of bytes per pixel
198  int PixelSize() const {return Channels() * ChannelSize(); }
199 
200  // Return depth type (e.g. IPL_DEPTH_8U, IPL_DEPTH_32F) which is the number
201  // of bits per channel and with the signed bit set.
202  // This is known at compile time using specializations.
203  int Depth() const;
204 
205  inline const T* Row(int r) const {
206  return reinterpret_cast<T*>(image_->imageData + r*image_->widthStep);
207  }
208 
209  inline T* Row(int r) {
210  return reinterpret_cast<T*>(image_->imageData + r*image_->widthStep);
211  }
212 
213  // Pixel accessors which returns a pointer to the start of the channel
214  inline T* operator() (int c, int r) {
215  return reinterpret_cast<T*>(image_->imageData + r*image_->widthStep) +
216  c*Channels();
217  }
218 
219  inline const T* operator() (int c, int r) const {
220  return reinterpret_cast<T*>(image_->imageData + r*image_->widthStep) +
221  c*Channels();
222  }
223 
224  // Copy the contents from another image which is just a convenience to cvCopy
225  void CopyFrom(const WImage<T>& src) { cvCopy(src.Ipl(), image_); }
226 
227  // Set contents to zero which is just a convenient to cvSetZero
228  void SetZero() { cvSetZero(image_); }
229 
230  // Construct a view into a region of this image
231  WImageView<T> View(int c, int r, int width, int height);
232 
233 protected:
234  // Disallow copy and assignment
235  WImage(const WImage&);
236  void operator=(const WImage&);
237 
238  explicit WImage(IplImage* img) : image_(img) {
239  assert(!img || img->depth == Depth());
240  }
241 
242  void SetIpl(IplImage* image) {
243  assert(!image || image->depth == Depth());
244  image_ = image;
245  }
246 
247  IplImage* image_;
248 };
