433  double operator ()(const Matx<_Tp, m, m>& a) const
434  {
435  Matx<_Tp, m, m> temp = a;
436  double p = LU(temp.val, m*sizeof(_Tp), m, 0, 0, 0);
437  if( p == 0 )
