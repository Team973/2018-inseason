68 template <typename B>
69 B Graph::walkBreadthFirst(int from, B body) const
70 {
71  std::vector<bool> was(numVertices(), false);
72  std::queue<int> vertices;
73 
74  was[from] = true;
75  vertices.push(from);
76 
77  while (!vertices.empty())
78  {
79  int vertex = vertices.front();
80  vertices.pop();
81 
82  std::list<GraphEdge>::const_iterator edge = edges_[vertex].begin();
83  for (; edge != edges_[vertex].end(); ++edge)
84  {
85  if (!was[edge->to])
86  {
87  body(*edge);
88  was[edge->to] = true;
89  vertices.push(edge->to);
90  }
91  }
92  }
93 
94  return body;
95 }
