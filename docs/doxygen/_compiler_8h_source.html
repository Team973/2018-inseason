39 # define LLVM_GNUC_PREREQ(maj, min, patch) \
40  ((__GNUC__ << 20) + (__GNUC_MINOR__ << 10) + __GNUC_PATCHLEVEL__ >= \
41  ((maj) << 20) + ((min) << 10) + (patch))
