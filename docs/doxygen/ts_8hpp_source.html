499  virtual void run_func(void) = 0;
500  int test_case_idx;
501 
502  template<class F>
503  int run_test_case( int expected_code, const string& _descr, F f)
504  {
505  int errcount = 0;
506  bool thrown = false;
507  const char* descr = _descr.c_str() ? _descr.c_str() : "";
508 
509  try
510  {
511  f();
512  }
513  catch(const cv::Exception& e)
514  {
515  thrown = true;
516  if( e.code != expected_code )
517  {
518  ts->printf(TS::LOG, "%s (test case #%d): the error code %d is different from the expected %d\n",
519  descr, test_case_idx, e.code, expected_code);
520  errcount = 1;
521  }
522  }
523  catch(...)
524  {
525  thrown = true;
526  ts->printf(TS::LOG, "%s (test case #%d): unknown exception was thrown (the function has likely crashed)\n",
527  descr, test_case_idx);
528  errcount = 1;
529  }
530  if(!thrown)
531  {
532  ts->printf(TS::LOG, "%s (test case #%d): no expected exception was thrown\n",
533  descr, test_case_idx);
534  errcount = 1;
535  }
536  test_case_idx++;
537 
538  return errcount;
539  }
540 };

648 class ifstream : public stringstream

650  FILE* f;
651 public:
652  ifstream(const char* filename, ios_base::openmode mode = ios_base::in)
653  : f(NULL)
654  {
655  string modeStr("r");
656  printf("Open file (read): %s\n", filename);
657  if (mode & ios_base::binary)
658  modeStr += "b";
659  f = fopen(filename, modeStr.c_str());
660 
661  if (f == NULL)
662  {
663  printf("Can't open file: %s\n", filename);
664  return;
665  }
666  fseek(f, 0, SEEK_END);
667  size_t sz = ftell(f);
668  if (sz > 0)
669  {
670  char* buf = (char*) malloc(sz);
671  fseek(f, 0, SEEK_SET);
672  if (fread(buf, 1, sz, f) == sz)
