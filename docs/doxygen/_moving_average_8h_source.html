28 class MovingAverage {

31  int _in;

33  int _cnt;

36  float * _d;
37 public:
38  MovingAverage(int capacity) {
39  _cap = capacity;
40  _d = new float[_cap];
41  Clear();
42  }
43  float Process(float input) {
44  Push(input);
45  return _sum / (float) _cnt;
46  }
47  void Clear() {
48  _in = 0;
49  _ou = 0;
50  _cnt = 0;
51 
52  _sum = 0;
53  }
54  void Push(float d) {
55  /* process it */
56  _sum += d;
57 
58  /* if full, pop one */
59  if (_cnt >= _cap)
60  Pop();
61 
62  /* push new one */
63  _d[_in] = d;
64  if (++_in >= _cap)
65  _in = 0;
66  ++_cnt;
67  }
68  void Pop() {
69  /* get the oldest */
70  float d = _d[_ou];
71 
72  /* process it */
73  _sum -= d;
74 
75  /* pop it */
76  if (++_ou >= _cap)
77  _ou = 0;
78  --_cnt;
79  }
80  //-------------- Properties --------------//
81  float GetSum() {
82  return _sum;
83  }
84  int GetCount() {
85  return _cnt;
86  }
87 };
