311  void ShapeUp()
312  {
313  std::vector<int> savedCode;
314  std::vector<cv::String> savedName;
315 
316  for(int i = 0; i < (int)implCode.size(); i++)
317  {
318  bool match = false;

320  {

322  {
323  match = true;

329  savedCode.push_back(implCode[i]);
