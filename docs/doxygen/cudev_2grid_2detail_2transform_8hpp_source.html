184  template <int SHIFT, typename SrcType, typename DstType, class UnOp, class MaskPtr>
185  __global__ void transformSmart(const GlobPtr<SrcType> src_, GlobPtr<DstType> dst_, const UnOp op, const MaskPtr mask, const int rows, const int cols)
186  {
187  typedef typename MakeVec<SrcType, SHIFT>::type read_type;
188  typedef typename MakeVec<DstType, SHIFT>::type write_type;
189 
190  const int x = blockIdx.x * blockDim.x + threadIdx.x;
191  const int y = blockIdx.y * blockDim.y + threadIdx.y;
192  const int x_shifted = x * SHIFT;
193 
194  if (y < rows)
195  {
196  const SrcType* src = src_.row(y);
197  DstType* dst = dst_.row(y);
198 
199  if (x_shifted + SHIFT - 1 < cols)
200  {
201  const read_type src_n_el = ((const read_type*)src)[x];
202  write_type dst_n_el = ((const write_type*)dst)[x];
203 
204  OpUnroller<SHIFT>::unroll(src_n_el, dst_n_el, op, mask, x_shifted, y);
205 
206  ((write_type*)dst)[x] = dst_n_el;
207  }
208  else
209  {
210  for (int real_x = x_shifted; real_x < cols; ++real_x)
211  {
212  if (mask(y, real_x))
213  dst[real_x] = op(src[real_x]);
214  }
215  }
216  }
217  }
