719 /*
720  * Squared Euclidean distance functor
721  */
722 template<class T>
723 struct CV_EXPORTS SL2
724 {
725  enum { normType = NORM_L2SQR };
726  typedef T ValueType;
727  typedef typename Accumulator<T>::Type ResultType;
728 
729  ResultType operator()( const T* a, const T* b, int size ) const
730  {
731  return normL2Sqr<ValueType, ResultType>(a, b, size);
732  }
733 };

735 /*
736  * Euclidean distance functor
737  */
738 template<class T>
739 struct CV_EXPORTS L2
740 {
741  enum { normType = NORM_L2 };
742  typedef T ValueType;
743  typedef typename Accumulator<T>::Type ResultType;
744 
745  ResultType operator()( const T* a, const T* b, int size ) const
746  {
747  return (ResultType)std::sqrt((double)normL2Sqr<ValueType, ResultType>(a, b, size));
748  }
749 };
