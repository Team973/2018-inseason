603 static NiFpga_Inline NiFpga_Status NiFpga_MergeStatus(
604  NiFpga_Status* const status,
605  const NiFpga_Status newStatus)
606 {
607  if (!status)
608  return NiFpga_Status_InvalidParameter;
609  if (NiFpga_IsNotError(*status)
610  && (*status == NiFpga_Status_Success || NiFpga_IsError(newStatus)))
611  *status = newStatus;
612  return *status;
613 }
