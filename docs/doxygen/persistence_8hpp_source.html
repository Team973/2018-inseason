822  template<typename _Tp, int numflag> class VecReaderProxy
823  {
824  public:
825  VecReaderProxy( FileNodeIterator* _it ) : it(_it) {}
826  void operator()(std::vector<_Tp>& vec, size_t count) const
827  {
828  count = std::min(count, it->remaining);
829  vec.resize(count);
830  for (size_t i = 0; i < count; i++, ++(*it))
831  read(**it, vec[i], _Tp());
832  }
833  private:
834  FileNodeIterator* it;
835  };

837  template<typename _Tp> class VecReaderProxy<_Tp, 1>
838  {
839  public:
840  VecReaderProxy( FileNodeIterator* _it ) : it(_it) {}
841  void operator()(std::vector<_Tp>& vec, size_t count) const
842  {
843  size_t remaining = it->remaining;
844  size_t cn = DataType<_Tp>::channels;
845  int _fmt = DataType<_Tp>::fmt;
846  char fmt[] = { (char)((_fmt >> 8)+'1'), (char)_fmt, '\0' };
847  size_t remaining1 = remaining / cn;
848  count = count < remaining1 ? count : remaining1;
849  vec.resize(count);
850  it->readRaw(fmt, !vec.empty() ? (uchar*)&vec[0] : 0, count*sizeof(_Tp));
851  }
852  private:
853  FileNodeIterator* it;
854  };

1106 template<typename _Tp> static inline
1107 FileStorage& operator << (FileStorage& fs, const _Tp& value)
1108 {
1109  if( !fs.isOpened() )
1110  return fs;
1111  if( fs.state == FileStorage::NAME_EXPECTED + FileStorage::INSIDE_MAP )
