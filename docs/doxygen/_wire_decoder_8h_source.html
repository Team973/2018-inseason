89  /* Reads a 16-bit word. */
90  bool Read16(unsigned int* val) {
91  const char* buf;
92  if (!Read(&buf, 2)) return false;
93  unsigned int v = (*reinterpret_cast<const unsigned char*>(buf)) & 0xff;
94  ++buf;
95  v <<= 8;
96  v |= (*reinterpret_cast<const unsigned char*>(buf)) & 0xff;
97  *val = v;
98  return true;
99  }
