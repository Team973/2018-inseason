54 template <typename THandle, typename TStruct, int16_t size,
55  HAL_HandleEnum enumValue>
56 THandle LimitedHandleResource<THandle, TStruct, size, enumValue>::Allocate() {
57  // globally lock to loop through indices
58  std::lock_guard<wpi::mutex> lock(m_allocateMutex);
59  for (int16_t i = 0; i < size; i++) {
60  if (m_structures[i] == nullptr) {
61  // if a false index is found, grab its specific mutex
62  // and allocate it.
63  std::lock_guard<wpi::mutex> lock(m_handleMutexes[i]);
64  m_structures[i] = std::make_shared<TStruct>();
65  return static_cast<THandle>(createHandle(i, enumValue, m_version));
66  }
67  }
68  return HAL_kInvalidHandle;
69 }

71 template <typename THandle, typename TStruct, int16_t size,
72  HAL_HandleEnum enumValue>
73 std::shared_ptr<TStruct>
74 LimitedHandleResource<THandle, TStruct, size, enumValue>::Get(THandle handle) {
75  // get handle index, and fail early if index out of range or wrong handle
76  int16_t index = getHandleTypedIndex(handle, enumValue, m_version);
77  if (index < 0 || index >= size) {
78  return nullptr;
79  }
80  std::lock_guard<wpi::mutex> lock(m_handleMutexes[index]);
81  // return structure. Null will propogate correctly, so no need to manually
82  // check.
83  return m_structures[index];
84 }

87  HAL_HandleEnum enumValue>
88 void LimitedHandleResource<THandle, TStruct, size, enumValue>::Free(
89  THandle handle) {
90  // get handle index, and fail early if index out of range or wrong handle
91  int16_t index = getHandleTypedIndex(handle, enumValue, m_version);
92  if (index < 0 || index >= size) return;
93  // lock and deallocated handle
94  std::lock_guard<wpi::mutex> allocateLock(m_allocateMutex);
95  std::lock_guard<wpi::mutex> handleLock(m_handleMutexes[index]);
96  m_structures[index].reset();
97 }
